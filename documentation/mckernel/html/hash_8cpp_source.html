11 unsigned int MurmurHash2 ( const void * key, int lt, unsigned int seed )
12 {
13  //'m' and 'r' are mixing constants generated offline.
14  //They're not really 'magic', they just happen to work well.
15 
16  const unsigned int m = 0x5bd1e995;
17  const int r = 24;
18 
19  //Initialize the hash to a 'random' value
20 
21  unsigned int h = seed ^ lt;
22 
23  //Mix 4 bytes at a time into the hash
24 
25  const unsigned char * data = (const unsigned char *)key;
26 
27  while(lt >= 4)
28  {
29  unsigned int k = *(unsigned int *)data;
30 
31  k *= m;
32  k ^= k >> r;
33  k *= m;
34 
35  h *= m;
36  h ^= k;
37 
38  data += 4;
39  lt -= 4;
40  }
41 
42  //Handle the last few bytes of the input array
43 
44  switch(lt)
45  {
46  case 3: h ^= data[2] << 16;
47  case 2: h ^= data[1] << 8;
48  case 1: h ^= data[0];
49  h *= m;
50  };
51 
52  //Do a few final mixes of the hash to ensure the last few
53  //bytes are well-incorporated.
54 
55  h ^= h >> 13;
56  h *= m;
57  h ^= h >> 15;
58 
59  return h;
60 }
