17  unsigned long p2 = ceil(log2(M_dn));
18  M_dnpg = 1UL << p2;

33

36

38  uniform_real_distribution<> urd(0, 1);
39  for (unsigned long c = 0; c < M_dn_D; ++c)
40  if(urd(seed) < 0.5)
41  dlv_B[c] = -1;
42  else
43  dlv_B[c] = 1;
44

47  p[z] = z % M_dnpg;

135 RBF_GAUSSIAN::RBF_GAUSSIAN(float* data, const unsigned long nv, const unsigned long dn, const unsigned long D, mt19937 seed, const float sigma)
136 :McKernel(data, nv, dn, D, seed, sigma)
137 {
138 
139  vector<float> scalar_C(M_D);
140 
141  float* dlv_G = &dl_G[0];
142  float* dlv_C = &dl_C[0];
143  float* sv_C = &scalar_C[0];
144 
145  //Generate G using N(0,1) and compute FROBENIUS Norm
146  normal_distribution<> nd(0, 1);
147  fill(scalar_C.begin(), scalar_C.end(), 0.0);
148  for (unsigned long c = 0; c < M_dn_D; ++c)
149  {
150  dlv_G[c] = nd(seed);
151  sv_C[c / M_dnpg] += dlv_G[c] * dlv_G[c];
152  }
153 
154  for(unsigned long c = 0; c < M_D; ++c)
155  sv_C[c] = 1.0 / sqrt(sv_C[c]);
156 
157  //Generate C using Chi Distribution
158  float sigma_factor = 1.0 / (sigma * sqrt(M_dnpg));
159  chi_squared_distribution<> csd(M_dnpg);
160  for (unsigned long k = 0; k < M_D; ++k)
161  for (unsigned long c = 0; c < M_dnpg; ++c)
162  dlv_C[c + k * M_dnpg] = sigma_factor * sv_C[k] * sqrt(csd(seed));
163 
164  dlv_G = NULL;
165  dlv_C = NULL;
166  sv_C = NULL;
167 }

170 RBF_MATERN::RBF_MATERN(float* data, const unsigned long nv, const unsigned long dn, const unsigned long D, mt19937 seed,
171  const float sigma, const unsigned long t):McKernel(data, nv, dn, D, seed, sigma)
172 {
173 
174  vector<float> scalar_C(M_D);
175 
176  float* dlv_G = &dl_G[0];
177  float* dlv_C = &dl_C[0];
178  float* sv_C = &scalar_C[0];
179 
180  uniform_real_distribution<> urd(0, 1);
181  normal_distribution<> nd(0, 1);
182 
183  //Generate G using N(0,1) and compute FROBENIUS Norm
184  fill(scalar_C.begin(), scalar_C.end(), 0.0);
185  for (unsigned long c = 0; c < M_dn_D; ++c)
186  {
187  dlv_G[c] = nd(seed);
188  sv_C[c / M_dnpg] += dlv_G[c] * dlv_G[c];
189  }
190 
191  for(unsigned long c = 0; c < M_D; ++c)
192  sv_C[c] = 1.0 / sqrt(sv_C[c]);
193 
194  //Generate C for MAT√âRN Kernel
195  float sigma_factor = 1.0 / (sigma * sqrt(M_dnpg));
196  for (unsigned long k = 0; k < M_D; ++k)
197  {
198  for (unsigned long c = 0; c < M_dnpg; ++c)
199  {
200  float norm_psi;
201 
202  //Draw t i.i.d. samples psi_c uniformly from Sd
203  vector<float> psi_n(M_dnpg);
204  for (unsigned long n = 0; n < t; ++n)
205  {
206 
207  vector<float> psi(M_dnpg);
208  norm_psi = 0.0;
209 
210  //Generate sample z from n-dimensional standard normal distribution
211  for(unsigned long r = 0; r < M_dnpg; ++r)
212  {
213  psi[r] = nd(seed);
214  norm_psi += psi[r] * psi[r];
215  }
216 
217  //Compute ||z||
218  norm_psi = sqrt(norm_psi);
219 
220  //Generate sample from uniform distribution U^1/n
221  float u = pow(urd(seed), (1.0/M_dn_D));
222 
223  //Return r * U^1/n * z/||z||
224  for(unsigned long m = 0; m < M_dnpg; ++m)
225  psi_n[m] += psi[m] * u / norm_psi;
226 
227  }
228 
229  //Compute L2 norm of psi
230  norm_psi = 0.0;
231  for(unsigned long r = 0; r < M_dnpg; ++r)
232  norm_psi += psi_n[r] * psi_n[r];
233 
234  //Assign to C_cc
235  dlv_C[c + k * M_dnpg] = sigma_factor * sv_C[k] * sqrt(norm_psi);
236  }
237  }
238 
239  dlv_G = NULL;
240  dlv_C = NULL;
241  sv_C = NULL;
242 }
