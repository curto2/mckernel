60 Bitmap::Bitmap(std::string filepath, Channels channels)
61 {
62  //Variables
63  int stbi_fmt = 0;
64  int fsize = 0;
65  std::string fmt = getfileformat(filepath);
66 
67  //Check file format
68  if (fmt == "jpg" || fmt == "jpeg" || fmt == "png" || fmt == "bmp" || fmt == "tga")
69  {
70 
71  switch (channels)
72  {
73  case Bitmap::MONO: stbi_fmt = STBI_grey; break;
74  case Bitmap::RGB: stbi_fmt = STBI_rgb; break;
75  case Bitmap::RGBA: stbi_fmt = STBI_rgb_alpha; break;
76  default:
77  ERROR("number of channels not supported");
78  return;
79  }
80 
81  //Open file
82  FILE* file = fopen(filepath.c_str(), "rb");
83  if (!file) {
84  ERROR("can't open the file");
85  return;
86  }
87 
88  //Get file size
89  fseek(file, 0L, SEEK_END);
90  fsize = (int)ftell(file);
91  fseek(file, 0L, SEEK_SET);
92 
93  //Read all file into filedata
94  stbi_uc* fileData = (stbi_uc*)malloc(sizeof(stbi_uc)* fsize);
95  fread(fileData, sizeof(stbi_uc)* fsize, 1, file);
96  fclose(file);
97 
98  //Get image data
99  m_data = stbi_load_from_memory(fileData, fsize, &m_width, &m_height, &m_channels, stbi_fmt);
100 
101  //Free filedata
102  if (fileData) free(fileData);
103 
104  //Check for stb_image fail
105  if (m_data == NULL) {
106  ERROR("can't get the image data");
107  printf("Details : %s\n", stbi_failure_reason());
108  return;
109  }
110  }
111  //Check file format
112  else if (fmt == "pgm")
113  {
114  /*
115  PGM file structure:
116  2 bytes P5
117  1 byte whitespace(blanks, TAB, CRs, LFs)
118  ? bytes width in ASCII
119  1 byte whitespace
120  ? bytes height in ACII
121  1 byte whitespace
122  ? bytes maximum gray value in ASCII (0 < x < 65536)
123  1 byte whitespace (usually newline)
124  w*h*k bytes data (k = 1 if maximum gray is <= 255 else k = 2)
125  */
126 
127  //Open file
128  FILE* file = fopen(filepath.c_str(), "rb");
129  if (!file) {
130  ERROR("can't open the file");
131  return;
132  }
133 
134  //First let's check the header P5 string
135  std::string p5;
136  p5 += getc(file);
137  p5 += getc(file);
138  if (p5 != "P5") {
139  ERROR("can't find P5 string in the header");
140  return;
141  }
142 
143  //Jump whitespace
144  getc(file);
145 
146  //Read width and height
147  m_width = m_height = -1;
148  fscanf(file, "%d %d", &m_width, &m_height);
149  if (m_width <= 0 || m_width > 4096 || m_height <= 0 || m_height > 4096) {
150  ERROR("invalid image size");
151  return;
152  }
153 
154  //Jump whitespace
155  getc(file);
156 
157  //Read maximum grey value
158  int maxgrey = -1;
159  fscanf(file, "%d", &maxgrey);
160 
161  //Jump whitespace
162  getc(file);
163 
164  //Free old data and allocate space for new data
165  m_data = (unsigned char*)malloc(sizeof(char) * (m_width * m_height));
166 
167  //Set channel information, PGM format support only mono channel
168  m_channels = 1;
169 
170  //Load data
171  int k = maxgrey > 255 ? 2 : 1;
172  double scale = 255.f / (double)maxgrey;
173  unsigned char cbuf[2];
174  cbuf[0] = cbuf[1] = 0;
175  int ibuf;
176  for (int y = 0; y < m_height; y++) {
177  for (int x = 0; x < m_width; x++) {
178  fread(&cbuf, sizeof(char), k, file);
179  ibuf = (double)(cbuf[0] + (cbuf[1] >> 6)) * scale;
180  m_data[y * m_width + x] = ibuf;
181  }
182  }
183 
184  //Close file
185  fclose(file);
186  }
187  else
188  {
189  ERROR("can't load image, unknown file format");
190  printf("Details: format:%s\n", fmt.c_str());
191  }
192 }

411 void Bitmap::rotate(const unsigned int center_x, const unsigned int center_y, const float angle, const long fill_color)
412 {
413  #define GETBYTE(n, byte_pos) (n >> (8 * byte_pos)) & 0xFF
414 
415  //Allocate temporary buffer
416  float* temp = new float[m_width * m_height * m_channels];
417  for (int x = 0; x < m_width; x++)
418  for (int y = 0; y < m_height; y++)
419  for (int c = 0; c < m_channels; c++)
420  temp[(y * m_width + x) * m_channels + c] = GETBYTE(fill_color, m_channels -1 -c);
421 
422  //Compute constants
423  const float radians = angle * 3.14159f/180.0;
424  const float a = cos(radians);
425  const float b = sin(radians);
426  const int xoffset = center_x - (center_x * a - center_y * b);
427  const int yoffset = center_y - (center_x * b + center_y * a);
428 
429  //Compute rotation in temporary buffer
430  // x' = x * cos(a) + y * sin(a) + xoffset
431  // y' = y * cos(a) - x * sin(a) + yoffset
432 
433  int nx, ny;
434  for (int x = 0; x < m_width; x++) {
435  for (int y = 0; y < m_height; y++) {
436  nx = x * a - y * b + xoffset;
437  ny = x * b + y * a + yoffset;
438  if (nx < 0 || nx >= m_width) continue;
439  if (ny < 0 || ny >= m_height) continue;
440  for (int c = 0; c < m_channels; c++) {
441  temp[(y * m_width + x) * m_channels + c] = m_data[(ny * m_width + nx) * m_channels + c];
442  }
443  }
444  }
445 
446  //Copy result to final tensor
447  for (int z = 0; z < m_width * m_height * m_channels; z++) m_data[z] = temp[z];
448 
449  //Free temporary buffer
450  delete[] temp;
451 }
