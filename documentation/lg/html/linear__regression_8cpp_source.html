83  void linear_regression::fit(Tensor_float inputs, const Tensor_float targets, const float starting_learningrate,
84  const unsigned int epochs, const bool verbose)
85  {
86  ensure(inputs.width() == (int)_input_size);
87  ensure(inputs.height() > 1 && inputs.depth() == 1);
88  ensure(targets.width() == (int)_output_size);
89  ensure(targets.height() == inputs.height());
90 
91 
92  float learning_rate = starting_learningrate;
93  double medium_error = 0;
94  for (unsigned int e = 0; e < epochs; e++) {
95 
96  medium_error = 0;
97 
98  std::vector<int> shuffle_idx(inputs.height());
99  for (int c = 0; c < (int)shuffle_idx.size(); c++) shuffle_idx[c] = c;
100  std::random_shuffle(shuffle_idx.begin(), shuffle_idx.end());
101 
102  for (int c = 0; c < (int)inputs.height(); c++) {
103 
104  //Compute learning rate
105  learning_rate = starting_learningrate * (1.f - (double)(e * inputs.height() + c) / (double)(epochs * inputs.height()));
106 
107  int indx = shuffle_idx[c];
108 
109  //Calculate output
110  predict(inputs.ptr(0, indx));
111 
112  //Calculate prediction errors
113  for (unsigned int z = 0; z < _output_size; z++) {
114  _errors[z] = targets.at(indx, z) - _outputs[z];
115  medium_error += fabs(_errors[z]);
116  }
117 
118  //Update weights
119  for (unsigned int z = 0; z < _output_size; z++) {
120  _bias[z] += _errors[z] * learning_rate;
121  for (unsigned int k = 0; k < _input_size; k++)
122  _weights.at(k, z) += _errors[z] * inputs.at(indx, k) * learning_rate;
123  }
124  }
125  if (verbose) printf("Epoch: %d Mean Error: %f\n", e, medium_error / (double)inputs.height());
126  learning_rate *= 0.7;
127  }
128  }
