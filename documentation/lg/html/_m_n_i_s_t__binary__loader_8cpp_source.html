32 int big_endian_to_small(int big_endian)
33 {
34  return ((big_endian >> 24) &0xff) | // move byte 3 to byte 0
35  ((big_endian << 8) &0xff0000) | // move byte 1 to byte 2
36  ((big_endian >> 8) &0xff00) | // move byte 2 to byte 1
37  ((big_endian << 24) &0xff000000); // byte 0 to byte 3
38 }

41 void MNIST_binary_loader::load_images(std::ifstream& file, std::vector< std::vector<unsigned char> >& images)
42 {
43  //Check first 32 bit magic number
44  int MSB_check;
45  file.read(reinterpret_cast<char*>(&MSB_check), sizeof(int));
46  MSB_check = big_endian_to_small(MSB_check);
47  assert(MSB_check == 2051);
48 
49  //Get number of images
50  int images_count;
51  file.read(reinterpret_cast<char*>(&images_count), sizeof(int));
52  images_count = big_endian_to_small(images_count);
53 
54  //Get images width and height
55  int images_width, images_height;
56  file.read(reinterpret_cast<char*>(&images_width), sizeof(int));
57  file.read(reinterpret_cast<char*>(&images_height), sizeof(int));
58  images_width = big_endian_to_small(images_width);
59  images_height = big_endian_to_small(images_height);
60  assert(images_width == 28 && images_height == 28);
61 
62  //Allocate memory for all images
63  images = std::vector< std::vector< unsigned char> >(images_count);
64  for (int c = 0; c < images_count; c++)
65  images[c] = std::vector< unsigned char >(images_width * images_height);
66 
67  //Read data for each image

69  file.read(reinterpret_cast<char*>(&images[c][0]), sizeof(unsigned char) * images_width * images_height);
