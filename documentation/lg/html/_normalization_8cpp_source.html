192  void Normalization::backprop(std::vector<Operation*> &inputs)
193  {
194  //Check for correct input size
195  ensure(inputs.size() == 1);
196 
197 #ifdef CUDA_BACKEND
198 
199  lg::cuda::normalization_backward(_errors.pointer(), inputs[0]->_errors.pointer(), _deviation.pointer(),
200  &_params.pointer()[0], &_params.pointer()[1], &_params.pointer()[2], _epsilon, _size);
201 
202 #else
203  //Shortcuts
204  Tensor_float &out_errors = inputs[0]->_errors;
205 
206  //Precompute some expressions
207  float sum_errors = 0.f;
208  float sum_errors_dev = 0.f;
209  for (int c = 0; c < _errors.size(); c++) {
210  sum_errors += _errors[c];
211  sum_errors_dev += _errors[c] * _deviation[c];
212  }
213 
214  //Calculate output errors
215  for (int c = 0; c < out_errors.size(); c++) {
216  out_errors[c] = 1.0 / (float)_size * _gamma / sqrt(_variance + _epsilon) * ((float)_size *
217  _errors[c] - sum_errors - _deviation[c] / (_variance + _epsilon) * sum_errors_dev);
218  }
219 
220  /*
221  dh = (1. / N) * gamma * (var + eps)**(-1. / 2.) * (N * dy - np.sum(dy, axis=0)
222  - (h - mu) * (var + eps)**(-1.0) * np.sum(dy * (h - mu), axis=0))
223  */
224 #endif
