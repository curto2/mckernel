134  void rotate(Tensor_float& t, int image_width, int image_height, int image_channels, float degrees)
135  {
136  //Check errors
137  ensure(t.size() == image_width * image_height * image_channels);
138 
139  //Allocate temporary buffer
140  Tensor_float temp(t.size());
141  temp.fill(0);
142 
143  //Compute constants
144  const float angle = degrees * 3.14159f/180.0;
145  const float a = cos(angle);
146  const float b = sin(angle);
147  const int wh = image_width/2.f;
148  const int hh = image_height/2.f;
149  const int xoffset = wh - (wh * a - hh * b);
150  const int yoffset = hh - (wh * b + hh * a);
151 
152  //Compute rotation in temporary buffer
153  // x' = x * cos(a) + y * sin(a) + xoffset
154  // y' = y * cos(a) - x * sin(a) + yoffset
155  int nx, ny;
156  for (int x = 0; x < image_width; x++) {
157  for (int y = 0; y < image_height; y++) {
158  nx = x * a - y * b + xoffset;
159  ny = x * b + y * a + yoffset;
160  if (nx < 0 || nx >= image_width) continue;
161  if (ny < 0 || ny >= image_height) continue;
162  for (int c = 0; c < image_channels; c++) {
163  temp[(y * image_width + x) + c * image_width * image_height] = t[(ny * image_width + nx) + c * image_width * image_height];
164  }
165  }
166  }
167 
168  //Copy result to final tensor
169  for (int z = 0; z < t.size(); z++) t[z] = temp[z];
170  }

183  for (int x = 0; x < image_width; x++) {

210  {
211  //Check errors

213

218  tmp.fill(0);

220  for (int x = 0; x < width; x++) {
221  for (int y = 0; y < height; y++) {
222  nx = cx + (x - cx) * scale;
223  ny = cy + (y - cy) * scale;
224  if (nx < 0 || nx >= width) continue;
225  if (ny < 0 || ny >= height) continue;
226  for (int c = 0; c < channel; c++) {
227  tmp[(y * width + x) * channel + c] = t[(ny * width + nx) * channel + c];
228  }
229  }
230  }
231

233  for (int z = 0; z < t.size(); z++) t[z] = tmp[z];
