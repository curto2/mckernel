165  void normalization_forward(float* inputs, float* deviation, float* normalized, float* outputs, float* variance, float* gamma, float* beta, float epsilon, int size)
166  {
167  //Calculate mean
168  double mean = 0;
169  for (int c = 0; c < size; c++)
170  mean += inputs[c];
171  mean /= (double)size;
172 
173  //Subtract mean vector to all inputs and calculate variance
174  *variance = 0;
175  for (int c = 0; c < size; c++) {
176  deviation[c] = inputs[c] - mean;
177  *variance += deviation[c] * deviation[c];
178  }
179  *variance /= (double)size;
180 
181  //Calculate normalized vector
182  for (int c = 0; c < size; c++) {
183  normalized[c] = deviation[c] / sqrt(*variance + epsilon);
184  outputs[c] = normalized[c] * *gamma + *beta;
185  }
186  }

189  void normalization_backward(float* errors, float* out_errors, float* deviation, float* variance, float* gamma, float* beta, float epsilon, int size)
190  {
191  //Pre-compute some expressions
192  float sum_errors = 0.f;
193  float sum_errors_dev = 0.f;
194  for (int c = 0; c < size; c++) {
195  sum_errors += errors[c];
196  sum_errors_dev += errors[c] * deviation[c];
197  }
198 
199  //Calculate output errors
200  for (int c = 0; c < size; c++) {
201  out_errors[c] = 1.0 / (float)size * *gamma / sqrt(*variance + epsilon) * ((float)size *
202  errors[c] - sum_errors - deviation[c] / (*variance + epsilon) * sum_errors_dev);
203  }
204  }

256  int inputx, inputy;
257  float tmp;
258 
259  //Let's do all the filters
260  for (int f = 0; f < input_count; f++) {
261  for (int y = 0; y < output_height; y++) {
262  for (int x = 0; x < output_width; x++) {
263 
264  //Get input X and Y
265  inputx = x * stride;
266  inputy = y * stride;
267 
268  //Check for bad borders
269  stopX = (inputx + filter_size > input_width) ? (inputx + filter_size) - inputx : filter_size;
270  stopY = (inputy + filter_size > input_height) ? (inputy + filter_size) - inputy : filter_size;
271

274  for (sx = 0; sx < stopX; sx++) {
275  for (sy = 0; sy < stopY; sy++) {

280  if (tmp > max) {

282  maxbuffer[outid] = f * input_width * input_height + input_width * (inputy + sy) + inputx + sx;
283  }
284  }
285  }
