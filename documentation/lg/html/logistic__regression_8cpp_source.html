82  const Tensor_float& logistic_regression::predict(const Tensor_float input)
83  {
84  ensure(input.width() == (int)_input_size);
85  ensure(input.height() == 1 && input.depth() == 1);
86 
87  //Fill outputs with bias
88  for (int k = 0; k < (int)_output_size; k++)
89  _outputs[k] = _bias[k];
90 
91  //Compute weighted sum
92  for (int c = 0; c < (int)_input_size; c++) {
93  if (input[c] == 0) continue; //exploit sparsity
94  for (int k = 0; k < (int)_output_size; k++) {
95  _outputs[k] += _weights.at(c, k) * input[c];
96  }
97  }
98 
99  //Apply sigmoid activation function
100  for (int k = 0; k < (int)_output_size; k++)
101  _outputs[k] = sigmoid(_outputs[k]);
102 
103  return _outputs;
104  }

142  const float logistic_regression::fit_single_sample(Tensor_float input, Tensor_float target, const float learningrate)
143  {
144  //Calculate output
145  predict(input);
146 
147  //Calculate prediction errors
148  float error = 0;
149  for (int z = 0; z < (int)_output_size; z++) {
150  _errors[z] = target[z] - _outputs[z];
151  error += fabs(_errors[z]);
152  }
153 
154  //Update weights
155  for (int z = 0; z < (int)_output_size; z++) {
156  _bias[z] += _errors[z] * learningrate;
157  for (int k = 0; k < (int)_input_size; k++)
158  _weights.at(k, z) += _errors[z] * sigmoid_derivative(_outputs[z]) * input[k] * learningrate;
159  }
160 
161  return error;
162  }
