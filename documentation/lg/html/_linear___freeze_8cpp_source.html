149  void Linear_Freeze::update_parameters(const float learningrate)
150  {
151  if (_fixed_parameters == true) return;
152 
153 #ifdef CUDA_BACKEND
154 
155  //Gradient clipping
156  if (_gradient_clipping != 0)
157  cuda::gradient_clipping(_deltas.pointer(), _deltas.size(), _gradient_clipping);
158 
159  //Update weights using gradients
160  cuda::linear_update_parameters(_weights.pointer(), _bias.pointer(), _deltas.pointer(),
161  learningrate, _input_size, _size);
162 
163  //Regularization penalty
164  if (_l1_regularization != 0) lg::wrn::l1_regularization(_weights, _l1_regularization, learningrate);
165  else if (_l2_regularization != 0) lg::wrn::l2_regularization(_weights, _l2_regularization, learningrate);
166 
167 #else
168 
169  //Gradient clipping
170  if (_gradient_clipping != 0) {
171  for (int c = 0; c < (int)_deltas.size(); c++) {
172  if (_deltas[c] > _gradient_clipping) _deltas[c] = _gradient_clipping;
173  else if(_deltas[c] < -_gradient_clipping) _deltas[c] = -_gradient_clipping;
174  }
175  }
176 
177  //Update weights using gradients
178  int d = 0;
179  for (int c = 0; c < _weights.width(); c++) {
180  for (int k = 0; k < _weights.height(); k++)
181  _weights.at(k, c) += _deltas[d++] * learningrate;
182  _bias[c] += _deltas[d++] * learningrate;
183  }
184 
185  //Regularization penalty
186  if (_l1_regularization != 0) lg::wrn::l1_regularization(_weights, _l1_regularization, learningrate);
187  else if (_l2_regularization != 0) lg::wrn::l2_regularization(_weights, _l2_regularization, learningrate);
188 
189 #endif
190  }

233  void Linear_Freeze::run(const Tensor_float input, bool accumulate)
234  {
235  if (_use_bias)
236  {
237  if (!accumulate)
238  {
239  //Reset outputs with bias
240  for (int c = 0; c < _outputs.size(); c++)
241  _outputs[c] = _bias[c];
242  }
243  else
244  {
245  //Add bias to output
246  for (int c = 0; c < _outputs.size(); c++)
247  _outputs[c] += _bias[c];
248  }
249  }
250  else
251  {
252  if (!accumulate)
253  {
254  for (int c = 0; c < _outputs.size(); c++)
255  _outputs[c] = 0;
256  }
257  }
258 
259  //Compute all inputs
260  int weight_index = 0;
261  for (int c = 0; c < input.size(); c++) {
262  if (input[c] == 0) continue;
263  weight_index = c * _outputs.size();
264  for (int k = 0; k < _outputs.size(); k++)
265  _outputs[k] += _weights[weight_index++] * input[c];
266  }
267  }

312  void Linear_Freeze::gradient_check()
313  {
314  //Parameters
315  const int size = 100;
316  const int input_size = 100;
317  const float epsilon = 10e-4;
318 
319  //Test node
320  Linear_Freeze node(size);
321  node.initialize(input_size);
322 
323  //Random input
324  Tensor_float input(size);
325  input.fill(0.5, 0.5);
326 
327  //Random target
328  Tensor_float target(size);
329  target.fill(0.5, 0.5);
330 
331  //Cost function
332  Cost costfun(Cost::SquaredError);
333 
334  //Computed numerical gradients
335  Tensor_float numgrad(node._weights.width(), node._weights.height());
336 
337  //For each parameter
338  for (int c = 0; c < node._weights.size(); c++) {
339  float init_param = node._weights[c];
340  node._weights[c] = init_param + epsilon;
341  node.run(input, false);
342  float lossPlus = costfun.getError(node._outputs, target);
343 
344  node._weights[c] = init_param - epsilon;
345  node.run(input, false);
346  float lossMinus = costfun.getError(node._outputs, target);
347 
348  numgrad[c] = (lossPlus - lossMinus) / (2.f * epsilon);
349 
350  node._weights[c] = init_param;
351  }
352 
353  //Compute gradients with backprop code
354  node.reset_deltas(0);
355  node.run(input, false);
356  costfun.getDelta(node._outputs, target, node._errors);
357  node.accumulate_deltas(input);
358 
359  int d = 0;
360  float max = 0;
361  double medium_delta_size = 0;
362  Tensor_float distances(numgrad.width(), numgrad.height());
363  for (int c = 0; c < node._weights.width(); c++) {
364  for (int k = 0; k < node._weights.height(); k++) {
365  medium_delta_size += fabs(node._deltas[d]);
366  distances.at(k, c) = fabs(numgrad.at(k, c) + node._deltas[d]);
367  if (distances.at(k, c) > max)
368  max = distances.at(k, c);
369  d++;
370  }
371  d++; //Bias
372  }
373  medium_delta_size /= node._deltas.size();
374 
375  const float tollerance = medium_delta_size * 0.05;
376  if (max > tollerance) printf("Gradient looks bad, differs by %f with medium_delta_size %f\n", max, medium_delta_size);
377  else printf("Gradient looks good, differs max by %f with medium_delta_size %f\n", max, medium_delta_size);
378 
379  //printf("%s\n", distances.tostring().c_str());
380  }
