380  void Convolution::update_parameters(const float learningrate)
381  {
382  if (_fixed_parameters) return;
383 
384 #ifdef CUDA_BACKEND
385 
386  //Gradient clipping
387  if (_gradient_clipping != 0)
388  cuda::gradient_clipping(_weights_delas.pointer(), _weights_delas.size(), _gradient_clipping);
389 
390  //Update weights using gradients
391  cudaconv.update_weights(_weights.pointer(), _weights_deltas.pointer(), _bias.pointer(),
392  _bias_deltas.pointer(), learningrate);
393 
394  //Regularization penalty
395  if (_l1_regularization != 0) lg::wrn::l1_regularization(_weights, _l1_regularization, learningrate);
396  else if (_l2_regularization != 0) lg::wrn::l2_regularization(_weights, _l2_regularization, learningrate);
397 
398 #else
399 
400  //Gradient clipping
401  if (_gradient_clipping != 0) {
402  for (int z = 0; z < (int)_weights_deltas.size(); z++) {
403  if (_weights_deltas[z] > _gradient_clipping) _weights_deltas[z] = _gradient_clipping;
404  else if(_weights_deltas[z] < -_gradient_clipping) _weights_deltas[z] = -_gradient_clipping;
405  }
406  }
407 
408  int deltas_index = 0;
409  int weights_index = 0;
410 
411  for (int f = 0; f < _filter_count; f++) {
412 
413  //Update weights
414  for (int k = 0; k < _input_count; k++) {
415 
416  //Update filter weights for this input
417  for (int w = 0; w < _filter_width * _filter_height; w++)
418  _weights[weights_index++] += _weights_deltas[deltas_index++] * learningrate;
419  }
420 
421  //Update bias
422  _bias[f] += _bias_deltas[f] * learningrate;
423  }
424 
425  //Regularization penalty
426  if (_l1_regularization != 0) lg::wrn::l1_regularization(_weights, _l1_regularization, learningrate);
427  else if (_l2_regularization != 0) lg::wrn::l2_regularization(_weights, _l2_regularization, learningrate);
428 
429 #endif
430  }

521  void Convolution::gradient_check()
522  {
523  //Parameters
524  const int input_width = 8;
525  const int input_height = 8;
526  const int input_depth = 3;
527  const double epsilon = 10e-4;
528 
529  //Test node
530  const int filter_size = 4;
531  const int filter_count = 10;
532  const int stride = 1;
533  const int padding = 1;
534  Convolution node(filter_size, filter_count, stride, padding);
535  node.initialize(input_width, input_height, input_depth);
536 
537  //Random input
538  Tensor_float input(input_width, input_height, input_depth);
539  input.fill(0.5, 0.5);
540 
541  //Random target
542  Tensor_float target(node._outputs.size());
543  target.fill(0.5, 0.5);
544 
545  //Cost function
546  Cost costfun(Cost::SquaredError);
547 
548  //Computed numerical gradients
549  Tensor_float numgrad(node._weights.width(), node._weights.height(), node._weights.depth());
550 
551  //For each parameter
552  for (int z = 0; z < node._weights.size(); z++) {
553  float init_param = node._weights[z];
554  node._weights[z] = init_param + epsilon;
555  node.run(input, false);
556  double lossPlus = costfun.getError(node._outputs, target);
557 
558  node._weights[z] = init_param - epsilon;
559  node.run(input, false);
560  double lossMinus = costfun.getError(node._outputs, target);
561 
562  numgrad[z] = (lossPlus - lossMinus) / (2.f * epsilon);
563 
564  node._weights[z] = init_param;
565  }
566 
567  //Compute gradients with backprop code
568  node.reset_deltas(0);
569  node.run(input, false);
570  costfun.getDelta(node._outputs, target, node._errors);
571  node.accumulate_deltas(input);
572
