223  void Recurrent::run(const Tensor_float input, bool accumulate)
224  {
225  //Update btt position
226  int btt_oldpos = _btt_pos;
227  _btt_pos = (_btt_pos + 1) % _btt_steps;
228 
229  //Store input
230  _mem_inputs.ptr(0, _btt_pos).set(input);
231 
232  //Calculate state output
233  _x._outputs.point(_mem_outputs.ptr(0, _btt_pos));
234  _x.run(_mem_inputs.ptr(0, _btt_pos), false);
235  _rec._outputs.point(_mem_outputs.ptr(0, _btt_pos));
236  _rec.run(_mem_outputs.ptr(0, btt_oldpos), true);
237  Tanh::forward(_mem_outputs.ptr(0, _btt_pos), _mem_outputs.ptr(0, _btt_pos));
238  _out._outputs.point(_outputs);
239  _out.run(_mem_outputs.ptr(0, _btt_pos), false);
240  }

243  void Recurrent::backprop(Tensor_float out_errors)
244  {
245  for (int k = 0; k < _btt_steps; k++)
246  _mem_tmp_errors.ptr(0, k).fill(0);
247 
248  //Propagate errors backward to memory state
249  _out._errors.point(_errors);
250  _out.backprop(_mem_tmp_errors.ptr(0, 0));
251  Tanh::backward(_mem_tmp_errors.ptr(0, 0), _mem_outputs.ptr(0, _btt_pos), _mem_tmp_errors.ptr(0, 0));
252 
253  //Unroll and backpropagate errors
254  for (int k = 0; k < _btt_steps-1; k++) {
255  int btt_id = _btt_pos - k - 1;
256  if (btt_id < 0) btt_id += _btt_steps;
257  _x._errors.point(_mem_tmp_errors.ptr(0, k));
258  _x.backprop(out_errors);
259  _rec._errors.point(_mem_tmp_errors.ptr(0, k));
260  _rec.backprop(_mem_tmp_errors.ptr(0, k + 1));
261  Tanh::backward(_mem_tmp_errors.ptr(0, k + 1), _mem_outputs.ptr(0, btt_id), _mem_tmp_errors.ptr(0, k + 1));
262  }
263  _x._errors.point(_mem_tmp_errors.ptr(0, _btt_steps-1));
264  _x.backprop(out_errors);
265  }

268  void Recurrent::accumulate_deltas(const Tensor_float input)
269  {
270  //Accumulate output deltas
271  _out._errors.point(_errors);
272  _out.accumulate_deltas(_mem_outputs.ptr(0, _btt_pos));
273 
274  //Accumulate memory deltas
275  for (int z = 0; z < _btt_steps-1; z++) {
276  int btt_id = (_btt_pos - z) % _btt_steps;
277  if (btt_id < 0) btt_id += _btt_steps;
278  int btt_id2 = (_btt_pos - z - 1) % _btt_steps;
279  if (btt_id2 < 0) btt_id2 += _btt_steps;
280  _x._errors.point(_mem_tmp_errors.ptr(0, z));
281  _x.accumulate_deltas(_mem_inputs.ptr(0, btt_id));
282  _rec._errors.point(_mem_tmp_errors.ptr(0, z));
283  _rec.accumulate_deltas(_mem_outputs.ptr(0, btt_id2));

288  void Recurrent::gradient_check()
289  {
290  //Parameters
291  const int size = 20;
292  const int input_size = 10;
293  const float epsilon = 10e-4;
294 
295  //Test node
296  Recurrent node(size);
297  node.initialize(input_size);
298 
299  //Random input
300  Tensor_float input(input_size);
301  input.fill(0.5, 0.5);
302 
303  //Out errors
304  Tensor_float out_errors(input_size);
305  out_errors.fill(0);
306 
307  //Random target
308  Tensor_float target(size);
309  target.fill(0.5, 0.5);
310 
311  //Cost function
312  Cost costfun(Cost::SquaredError);
313 
314  //Computed numerical gradients
315  Tensor_float& node_weights = node._x._weights;
316  Tensor_float numgrad(node_weights.width(), node_weights.height(), node_weights.depth());
317 
318  //For each parameter
319  for (int z = 0; z < node_weights.size(); z++) {
320  float init_param = node_weights[z];
321  node_weights[z] = init_param + epsilon;
322  node.reset_hidden_state();
323  node.run(input, false);
324  float lossPlus = costfun.getError(node._outputs, target);
325 
326  node_weights[z] = init_param - epsilon;
327  node.reset_hidden_state();
328  node.run(input, false);
329  float lossMinus = costfun.getError(node._outputs, target);
330 
331  numgrad[z] = (lossPlus - lossMinus) / (2.f * epsilon);
332 
333  node_weights[z] = init_param;
334  }
335 
336  //Compute gradients with backprop code
337  node._x.reset_deltas(0);
338  node.reset_hidden_state();
339  node.run(input, false);
340  costfun.getDelta(node._outputs, target, node._errors);
341  node.backprop(out_errors);
342  node.accumulate_deltas(input);
343 
344  int d = 0;
345  float max = 0;
346  Tensor_float distances(numgrad.width(), numgrad.height(), numgrad.depth());
347  for (int z = 0; z < node_weights.width(); z++) {
348  for (int k = 0; k < node_weights.height(); k++) {
349  distances.at(k, z) = fabs(numgrad.at(k, z) + node._x._deltas[d]);
350  if (distances.at(k, z) > max)
351  max = distances.at(k, z);
352  d++;
353  }
354  d++; //Bias
355  }
356 
357  const float tollerance = 3 * 1e-3;
358  if (max > tollerance) printf("Gradient differs by %f\n", max);
359  else printf("Gradient looks good\n");
360 
361  printf("%s\n", distances.tostring().c_str());
362 
363  }
