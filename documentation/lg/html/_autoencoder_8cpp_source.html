205  void Autoencoder::run(const Tensor_float input, bool accumulate, bool training)
206  {
207 
208 
209  //Apply bias
210  for (int c = 0; c < _outputs.size(); c++)
211  _outputs[c] = _bias[c];
212 
213  if (training && _noise)
214  {
215  _noise_mask.fill(0, _noise);
216 
217  //Linear output calculation
218  int weight_index = 0;
219  for (int c = 0; c < input.size(); c++) {
220  if (input[c] == 0) continue;
221  weight_index = c * _outputs.size();
222  for (int k = 0; k < _outputs.size(); k++)
223  _outputs[k] += _weights[weight_index++] * (input[c] + _noise_mask[c]);
224  }
225  }
226  else
227  {
228  //Linear output calculation
229  int weight_index = 0;
230  for (int c = 0; c < input.size(); c++) {
231  if (input[c] == 0) continue;
232  weight_index = c * _outputs.size();
233  for (int k = 0; k < _outputs.size(); k++)
234  _outputs[k] += _weights[weight_index++] * input[c];
235  }
236  }
237 
238  //Apply activation function
239  lg::Relu::forward(_outputs, _outputs);
240 
241  //Store activations
242  int p = 0;
243  for (int c = 0; c < _size; c++)
244  if (_outputs[c] > 0) _activations[p++] = c;
245  _activations[p] = -1;
246  }

255  void Autoencoder::accumulate_deltas(const Tensor_float input)
256  {
257  if (_fixed_parameters) return;
258 
259  _prediction.fill(0);
260  int out_index = 0;
261  for (int c = 0; c < (int)_activations.size(); c++) {
262  if (_activations[c] == -1) break;
263  out_index = _activations[c];
264  for (int k = 0; k < _input_size; k++)
265  _prediction[k] += _weights.at(k, out_index) * _outputs[out_index];
266  }
267 
268  //Compute prediction error
269  _error = 0;
270  for (int c = 0; c < _input_size; c++) {
271  _prediction_error[c] = input[c] - _prediction[c];
272  _error += fabs(_prediction_error[c]);
273  }
274 
275  //Calculate hidden errors
276  out_index = 0;
277  for (int c = 0; c < (int)_activations.size(); c++) {
278  if (_activations[c] == -1) break;
279  out_index = _activations[c];
280  for (int k = 0; k < _input_size; k++) {
281  _hidden_errors[out_index] += _weights.at(k, out_index) * _prediction_error[k]; //bias
282  }
283  }
284  lg::Relu::backward(_hidden_errors, _outputs, _hidden_errors);
285 
286  //Compute deltas
287  out_index = 0;
288  for (int c = 0; c < (int)_activations.size(); c++) {
289  if (_activations[c] == -1) break;
290  out_index = _activations[c];
291  for (int k = 0; k < _input_size; k++) {
292  _w_deltas.at(k, out_index) += _outputs[out_index] * _prediction_error[k]; //weights decoder
293  _w_deltas.at(k, out_index) += input[k] * _hidden_errors[out_index]; //weights encoder
294  }
295  _b_deltas[out_index] += _hidden_errors[out_index];
296  }
297 
298  //Reset errors
299  _hidden_errors.fill(0);
300  }
