92  ensure(inputs[0]->_outputs.depth() >= 1);

94  _input_count = inputs[0]->_outputs.depth();
95  _input_height = inputs[0]->_outputs.height();
96  _input_width = inputs[0]->_outputs.width();
97  }
98 
99  //Check that inputs have correct size
100  ensure(inputs[0]->_outputs.size() == _input_width * _input_height * _input_count);
101 
102  //Compute characteristics
103  _input_size = _input_width * _input_height * _input_count;
104  _output_width = (_input_width - _filter_size) / _stride + 1.0;
105  _output_height = (_input_height - _filter_size) / _stride + 1.0;
106  _output_size = _output_width * _output_height;
107  _size = _output_size * _input_count;
108 
109  //Initialize variables and buffers
110  _outputs.setshape(_output_width, _output_height, _input_count);
